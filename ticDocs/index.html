/* 'Variables used and there uses' */


let player = true; /* It is boolean type variable */

/* Used this variable for checking the current user. */
/* If value of player is {true} then the chance is for 'X' and {false} then chance for 'O' */

const box = document.getElementsByClassName("button");

/* It stores the HTMLCollection for all the classes with name 'button' */
/* Purpose of using this to handle 'click' event on each button with addEventListener */

const current = document.getElementById("status");

/* It is used to dynamically update the value of current user in the page */

let winner = false; /* Used to check there is a winner or not in the current game */

/* Value {false} means no winner, if value is {true} then there is a winner in the game */

let moveCount = 0; /* To count the number of moves completed */
/* It will increment by 1 after everymove. */

const winningCombinations = [ /* This is a 2D Array that stores all the winning combinations. */
  [0, 1, 2],
  [3, 4, 5],
  [6, 7, 8],
  [0, 3, 6],
  [1, 4, 7],
  [2, 5, 8],
  [0, 4, 8],
  [2, 4, 6],
];


let moves = [];/* It is an array that stores the moves of the game */

/* Loop is used to assign 9 value to 'null' in the moves array */
for (let i = 0; i < 9; i++) {
  moves[i] = null;
}
  

            
/* 'Functions used and there uses' */

/* "start()" function explanation */

/* This function is load when the body of the html load */
/* `body onload='start()'` as it is mention here*/

/* Use for this function is to create eventListener to all 9 buttons in the code */

/* Loop is used for adding eventListener one by one */

/* here 'box[i]' gets the element with class one by one */
/* Eg. 'box[0]' refers to the first element with classname 'button' */

/* addEventListener(arg1,arg2). It takes two arguments.*/
/* arg1 #First Argument is event name, eg. 'click','onload',etc. */
/* arg2 #Second Argument takes a function */

/* here event name is 'click'*/
/* function is () => { play(i) } */

/* it is adding a function play(i) to every element with classname 'button' */
/* in play(i), 'i' is an argument which is the element number from the above loop */
/* we pass 'i' as argument to check which button is clicked */

function start() {
  for (let i = 0; i < 9; i++) {
    box[i].addEventListener("click", () => {
      play(i);
    });
  }
}

/* This function called everytime a buttom is clicked */
function play(button) {
  moveCount++; /* increment the moveCount on every click */ 

  /* If player is "X" means value of player is (true) */
  if (player) {
    box[button].innerHTML = "X";/* Changing the text for the button clicked */
    box[button].style.color = "#12e177"; /* Styling the text color with js */
    current.innerHTML = "O"; /* Dispaying the next player name */
    current.style.color = "#ffc72a"; /* Displaying the color of current player name */
    moves[button] = "X"; /* Adding 'X' at the button value in the moves array */

    /* Eg if button = 3 then moves array is [null,null,null,"X",null,null,null,null,null] */
  } 
  /* This will run when player is 'O' or player value is (false) */
  else {
    box[button].innerHTML = "O"; /* Changing the text for the button clicked */
    box[button].style.color = "#ffc72a"; /* Styling the text when there is 'O' */
    current.innerHTML = "X"; /* Displaying the next player name */
    current.style.color = "#12e177"; /* Styling the  next player name */
    moves[button] = "O"; /* Adding 'O' at button value in the moves array */
  }
  player = !player; /* Changing player after every move */
  box[button].disabled = true; /* disabled the button which is clicked to prevent multiple clicking */
  box[button].classList.add("animate"); /* Adding class to button for the animation */

  checkWinner(); /* Calling checkWinner function everytime to check the winner */
}

/* This is a checkWinner function that check the winner for the game */
function checkWinner() {
  /* This loop run over all values in the 2D array winningCombinations */
  /* Eg. winningCombinations[0] = [0,1,2]; this gives an array with three values in it */


  for (let i = 0; i < 8; i++) {
    let a = winningCombinations[i][0]; /* Storing first value in variable 'a' */
    let b = winningCombinations[i][1]; /* Storing second value in variable 'b' */
    let c = winningCombinations[i][2]; /* Storing third value in variable 'c' */

    /* winner is when at loctions a,b,c in the array moves values are equal */
    /* OR moves[a]=moves[b]=moves[c] */

    /* In if statement we are checking the winning condition */
    /* Condition is true when value at (moves[a] is not null) AND (value at location a and b is equal) AND (value at location b and c is equal)


    if (moves[a] && moves[a] === moves[b] && moves[b] === moves[c]) {

      /* Executed whenever there is winner */
      /* Winner store that winner is "X" or "O" according to the value in moves[a] */

      winner = moves[a];

      /* We will call stop function to finish the game */

      stop(winner, a, b, c); /* We pass 4 arguments in the stop function */  
      /* now assgining winner true */

      winner = true;
    }
    /* We are using this if statement to show there is tie between the players */

    /* Executed when there is no winner and moveCount>8 means no button is empty */
    if (!winner && moveCount > 8) {
      /* We will remove the next chance status now by display ='none' property */

      document.getElementById("game").style.display = "none";

      /* To show text that there is a tie */

      document.getElementById(
        "winner"
      ).innerHTML = `X and O Tied`;
    }
  }
}
/* Stop function is executed when there is a winner */

function stop(winner, a, b, c) {
  /* here a,b,c gives the winning combinations for the current game */
  let color = "#12e177";
  /* We will remove the next chance status now by display ='none' property */
  document.getElementById("game").style.display = "none";
  /* we are adding win class to the winning combination */
  box[a].classList.add("win");
  box[b].classList.add("win");
  box[c].classList.add("win");

  /* End of the game to disabled all the button on the board */
  for (let i = 0; i < 9; i++) {
    box[i].disabled = true;
  }
  /* Setting color for the winner */
  if (winner === "O") {
    color = "#ffc72a";
  }
  /* Displaying the winner on the screen */
  document.getElementById("winner").innerHTML =
    `Winner is ` + winner + ``;
}
